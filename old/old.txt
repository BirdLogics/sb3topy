    def __init__(self, parent={}, isOriginal=True):
        if isOriginal:
            # Gets either the value in parent or a default value
            default = lambda k, d: k in parent and parent[k] or d

            self.x = default("x", 0)
            self.y = default("y", 0)
            self.size = default("size", 100)
            self.direction = default("direction", 90)
            self.draggable = default("draggable", True)
            self.currentCostume = default("currentCostume", 0)
            self.visible = default("visible", True)
            self.rotationStyle = default("rotationStyle", "all around")

        else:
            self.variables = parent.variables.copy()
            self.lists = []
            for name, list in parent.lists.items():
                self.lists[name] = list.copy()

            self.costume = parent.costumes[self.currentCostume]
            self.costumeCount = len(self.costumes)
